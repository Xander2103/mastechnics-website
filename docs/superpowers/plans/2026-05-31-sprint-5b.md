## Task 2 — Validation error recovery

**Files:**
- Modify: `resources/views/pages/partials/request-page.blade.php`
  - `@php` block: add `$hasErrors` / `$errorFieldNames`
  - Form HTML: add hidden error-state element, add `data-fields` to each form section
  - Inline `<script>`: add `findErrorStep()` helper, update `init()`

### How it works

1. PHP renders `<div id="wizardErrorState" data-has-errors="true|false" data-error-fields='[...]'>` with the list of Laravel error keys.
2. Each `<section class="form-section">` gets a `data-fields="field1,field2"` attribute derived from the step config. The `service_category_selection` step uses `service_category`; `airco_rooms` step uses `rooms`; other steps use their field names.
3. JS `findErrorStep(sections, errorFields)` walks visible sections and returns the index of the first section whose `data-fields` intersects any error key (supports prefix matching for array fields like `rooms.0.type`).
4. `init()` calls `findErrorStep` and passes the result to `showStep()` instead of always starting at 0.

```js
    // ── Validation error recovery ─────────────────────────────────────────────
    function findErrorStep(sections, errorFields) {
        if (!errorFields || errorFields.length === 0) return 0;
        for (var i = 0; i < sections.length; i++) {
            var sectionFields = (sections[i].dataset.fields || '').split(',').filter(Boolean);
            for (var f = 0; f < sectionFields.length; f++) {
                var fieldName = sectionFields[f];
                for (var e = 0; e < errorFields.length; e++) {
                    // Exact match OR prefix match for array fields (e.g. "rooms" matches "rooms.0.type")
                    if (errorFields[e] === fieldName || errorFields[e].indexOf(fieldName + '.') === 0) {
                        return i;
                    }
                }
            }
        }
        return 0;
    }

    // ── Init ──────────────────────────────────────────────────────────────────
    function init() {
        formCard.classList.add('is-wizard-active');
        var initialCategory = getSelectedCategory();
        visibleSections     = computeVisible(initialCategory);
        updateConditionalVisibility(initialCategory);

        var errorEl   = document.getElementById('wizardErrorState');
        var hasErrors = errorEl && errorEl.dataset.hasErrors === 'true';
        var errorFields = [];
        if (hasErrors && errorEl.dataset.errorFields) {
            try { errorFields = JSON.parse(errorEl.dataset.errorFields); } catch (e) {}
        }

        var startIndex = hasErrors ? findErrorStep(visibleSections, errorFields) : 0;
        showStep(visibleSections, startIndex);
    }
```

```js
    // ── Room manager ──────────────────────────────────────────────────────────
    function initRoomManager() {
        var manager = document.getElementById('roomManager');
        if (!manager) return; // Not on an airco_offerte page

        var addBtn = document.getElementById('addRoomBtn');

        function getEntries() {
            return Array.from(manager.querySelectorAll('.room-entry'));
        }

        function renumberEntries() {
            var roomLabel = manager.dataset.roomLabel || 'Kamer';
            getEntries().forEach(function (entry, idx) {
                entry.dataset.roomIndex = idx;

                // Update title via textContent (safe — idx is an integer)
                var title = entry.querySelector('.room-entry-title');
                if (title) {
                    title.textContent = roomLabel + ' ' + (idx + 1);
                }

                // Update all input/select names rooms[N][field]
                entry.querySelectorAll('[name]').forEach(function (el) {
                    el.name = el.name.replace(/rooms\[\d+\]/, 'rooms[' + idx + ']');
                });

                // Show remove button only for rooms after the first
                var removeBtn = entry.querySelector('.room-remove-btn');
                if (removeBtn) {
                    removeBtn.style.display = idx === 0 ? 'none' : '';
                }
            });
        }

        function calcSurface(entry) {
            var w = parseFloat(entry.querySelector('.room-width-input').value) || 0;
            var l = parseFloat(entry.querySelector('.room-length-input').value) || 0;
            var surf = entry.querySelector('.room-surface-input');
            if (surf) {
                surf.value = (w > 0 && l > 0) ? (Math.round(w * l * 10) / 10) : '';
            }
        }

        // Surface auto-calc on input change (delegated to room entries)
        manager.addEventListener('input', function (event) {
            var target = event.target;
            if (target.classList.contains('room-width-input') || target.classList.contains('room-length-input')) {
                var entry = target.closest('.room-entry');
                if (entry) calcSurface(entry);
            }
        });

        // Remove room (delegated)
        manager.addEventListener('click', function (event) {
            var removeBtn = event.target.closest('.room-remove-btn');
            if (!removeBtn) return;
            var entry = removeBtn.closest('.room-entry');
            if (!entry) return;
            if (getEntries().length <= 1) return; // Never remove the last room
            entry.remove();
            renumberEntries();
        });

        // Add room
        if (addBtn) {
            addBtn.addEventListener('click', function () {
                var entries = getEntries();
                var last    = entries[entries.length - 1];
                var clone   = last.cloneNode(true);
                var newIdx  = entries.length;

                // Clear all input/select values in the clone
                clone.querySelectorAll('input, select').forEach(function (el) {
                    el.value = '';
                });

                // Set correct names (rooms[N][field])
                clone.querySelectorAll('[name]').forEach(function (el) {
                    el.name = el.name.replace(/rooms\[\d+\]/, 'rooms[' + newIdx + ']');
                });

                // Set title via textContent (safe — newIdx is an integer, roomLabel is from data attribute)
                var roomLabel = manager.dataset.roomLabel || 'Kamer';
                var title = clone.querySelector('.room-entry-title');
                if (title) {
                    title.textContent = roomLabel + ' ' + (newIdx + 1);
                }

                // Show remove button on the new room
                var removeBtn = clone.querySelector('.room-remove-btn');
                if (removeBtn) {
                    removeBtn.style.display = '';
                    // Update remove button label from data attribute
                    var removeLabel = manager.dataset.roomRemoveLabel || 'Verwijder kamer';
                    removeBtn.textContent = removeLabel;
                }

                clone.dataset.roomIndex = newIdx;
                manager.insertBefore(clone, addBtn);
            });
        }

        // Init: show/hide remove buttons for server-rendered rooms
        renumberEntries();

        // Init: calculate surface for pre-filled rooms (old input after validation error)
        getEntries().forEach(function (entry) { calcSurface(entry); });
    }
```
